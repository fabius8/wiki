* vlan 配置
```
typ:isadmin>configure>bridge>port>1/1/2/3/99/8/1# info
configure bridge
#----------------------------------------------------------------------------------------------------------------------------------------------
echo "bridge"
#----------------------------------------------------------------------------------------------------------------------------------------------
port 1/1/2/3/99/8/1
  max-unicast-mac 1000
  vlan-id 10
    tag single-tagged
    l2fwder-vlan 4000
    vlan-scope local
  exit
  vlan-id 3000
  exit
  pvid 3000
exit
```
